148 def run(self):
149 if not self.on:
150 return 0
151
152 # Soft-start reference ramping
153 if self.num_corrections > 0:
154 self.r_u = self.r * ((self.total_num_corrections - self.num_corrections)
155 / self.total_num_corrections)
156 self.num_corrections -= 1
157 else:
158 self.r_u = self.r
159
160 # Get current measured value
161 self.x_h = self.sensor.get_data()
162
163 # Error update
164 self.prev_error = self.e
165 self.e = self.r_u - self.x_h
166
167 # Compute timing delta in seconds
168 time_delta = self.sensor.dt / 1e6
169 if self.first_run:
170 time_delta = 0.020 # Approx. first-cycle assumption
171
172 # Integral and derivative updates
173 if self.Ki:
174 self.integral += self.e * time_delta
175 if self.Kd:
176 self.derivative = (self.e - self.prev_error) / time_delta
177
178 # Battery droop compensation
179 if self.battery is not None:
180 Kdroop = 1 / self.battery.get_cur_perc()
181 else:
182 Kdroop = 1
183
184 # Controller output
185 self.a = Kdroop * (
186 (self.Kp * self.e) +
187 (self.integral * self.Ki) +
188 (self.derivative * self.Kd) +
189 ((self.r_u * self.Kff) + copysign(1, self.r) * self.PWM_start)
190 )
191
192 # Saturation
193 self.astar = min(self.max_min, max(-self.max_min, self.a))
194
195 # Anti-windup correction
196 if self.Kw:
197 self.integral -= (self.a - self.astar) * self.Kw
198
199 return self.astar
