141 def schedule(self) -> bool:
142 if self.ready():
143
144 # Reset the go flag for the next run
145 self.go_flag = False
146
147 # If profiling, save the start time
148 if self._prof:
149 stime = utime.ticks_us()
150
151 # Run the method belonging to the state which should be run next
152 curr_state = next(self._run_gen)
153
154 # If profiling or tracing, save timing data
155 if self._prof or self._trace:
156 etime = utime.ticks_us()
157
158 # If profiling, save timing data
159 if self._prof:
160 self._runs += 1
161 runt = utime.ticks_diff(etime, stime)
162 if self._runs > 2:
163 self._run_sum += runt
164 if runt > self._slowest:
165 self._slowest = runt
166
167 # If transition logic tracing is on, record a transition; if not,
168 # ignore the state. If out of memory, switch tracing off and
169 # run the memory allocation garbage collector
170 if self._trace:
171 try:
172 if curr_state != self._prev_state:
173 self._tr_data.append((utime.ticks_diff(etime, self._prev_time), curr_state))
174 except MemoryError:
175 self._trace = False
176 gc.collect()
177
178 self._prev_state = curr_state
179 self._prev_time = etime
180
181 return True
182
183 else:
184 return False

304 def append(self, task):
305 # See if there's a tasklist with the given priority in the main list
306 new_pri = task.priority
307 for pri in self.pri_list:
308 # If a tasklist with this priority exists, add this task to it.
309 if pri[0] == new_pri:
310 pri.append(task)
311 break
312
313 # If the priority isn't in the list, this else clause starts a new
314 # priority list with this task as first one. A priority list has the
315 # priority as element 0, an index into the list of tasks (used for
316 # round-robin scheduling those tasks) as the second item, and tasks
317 # after those
318 else:
319 self.pri_list.append([new_pri, 2, task])
320
321 # Make sure the main list (of lists at each priority) is sorted
322 self.pri_list.sort(key=lambda pri: pri[0], reverse=True)

346 def pri_sched(self):
347 # Go down the list of priorities, beginning with the highest
348 for pri in self.pri_list:
349 # Within each priority list, run tasks in round-robin order
350 # Each priority list is [priority, index, task, task, ...] where
351 # index is the index of the next task in the list to be run
352 tries = 2
353 length = len(pri)
354 while tries < length:
355 ran = pri[pri[1]].schedule()
356 tries += 1
357 pri[1] += 1
358 if pri[1] >= length:
359 pri[1] = 2
360 if ran:
361 return
