93 def _write_reg(self, reg: tuple, input: str):
94 # Writes a string input to the IMU: ie input = '01110xxx'
95 # This write functionality can work on 8 or 16 bits
96 # ASSUMED THAT CONFIGMODE IS SET APPROPRIATELY BEFORE CALLING THIS FUNCTION
97
98 # determine number of bytes in register location
99 length = calcsize(reg[1])
100 if length > 2 or len(input) // 8 != length:
101 raise ValueError("Write to register is not the correct length")
102
103 # Create and mask and or mask
104 and_mask = 0b1111_1111 if length == 1 else 0b1111_1111_1111_1111
105 or_mask = 0b0000_0000 if length == 1 else 0b0000_0000_0000_0000
106 for char_index in range(len(input)):
107 char = input[-char_index - 1]
108 if char == "x" or char == "X":
109 continue # and mask is already set, or mask is already cleared
110 elif char == "0":
111 and_mask &= ~(1 << (char_index)) # clear the and bit
112 elif char == "1":
113 and_mask &= ~(1 << (char_index)) # clear the and bit
114 or_mask |= 1 << (char_index) # set the or bit
115 else:
116 raise ValueError("mask chars must be '0','1','x', or 'X'")
117
118 self.i2c.mem_write(
119 ((self._read_reg(reg)[0] & and_mask) | or_mask),
120 self.DEV_ADDR,
121 reg[0],
122 addr_size=(8 if length == 1 else 16),
123 )
