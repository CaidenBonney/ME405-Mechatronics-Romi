102 def _read_reg(self, reg: tuple):
103 # determine number of bytes from reg read code
104 length = calcsize(reg[1])
105
106 # Create a memoryview object of the right size
107 buf = memoryview(self._buf)[:length]
108
109 # Read the the I2C bytes into memoryview
110 self.i2c.mem_read(buf, self.DEV_ADDR, reg[0])

112 return unpack_from(reg[1], buf)

118 def _write_reg(self, reg: tuple, input: str):
119 # Writes a string input to the IMU: ie input = '01110xxx'
120 # This write functionality can work on 8 or 16 bits
121 # ASSUMED THAT CONFIGMODE IS SET APPROPRIATELY BEFORE CALLING THIS FUNCTION
122
123 # determine number of bytes in register location
124 length = calcsize(reg[1])
125 if length > 2 or len(input) // 8 != length:
126 raise ValueError("Write to register is not the correct length")
127
128 # Create and mask and or mask
129 and_mask = 0b1111_1111 if length == 1 else 0b1111_1111_1111_1111
130 or_mask = 0b0000_0000 if length == 1 else 0b0000_0000_0000_0000
131 for char_index in range(len(input)):
132 char = input[-char_index - 1]
133 if char == "x" or char == "X":
134 continue # and mask is already set, or mask is already cleared
135 elif char == "0":
136 and_mask &= ~(1 << (char_index)) # clear the and bit
137 elif char == "1":
138 and_mask &= ~(1 << (char_index)) # clear the and bit
139 or_mask |= 1 << (char_index) # set the or bit
140 else:
141 raise ValueError("mask chars must be '0','1','x', or 'X'")
142
143 self.i2c.mem_write(
144 ((self._read_reg(reg)[0] & and_mask) | or_mask),
145 self.DEV_ADDR,
146 reg[0],
147 addr_size=(8 if length == 1 else 16),
148 )

157 def set_mode(self, mode: int):
158 if mode == 0:
159 self._write_reg(IMU.reg.OPR_MODE, "xxxx0000") # 0: CONFIGMODE
160 elif mode == 1:
161 self._write_reg(IMU.reg.OPR_MODE, "xxxx1000") # 1: IMU
162 elif mode == 2:
163 self._write_reg(IMU.reg.OPR_MODE, "xxxx1001") # 2: COMPASS
164 elif mode == 3:
165 self._write_reg(IMU.reg.OPR_MODE, "xxxx1010") # 3: M4G
166 elif mode == 4:
167 self._write_reg(IMU.reg.OPR_MODE, "xxxx1011") # 4: NDOF_FMC_OFF
168 elif mode == 5:
169 self._write_reg(IMU.reg.OPR_MODE, "xxxx1100") # 5: NDOF
170 else:
171 raise ValueError("Invalid mode")

185 def calibrate(self) -> bool:
186 if "IMU_cal.txt" in listdir():
187 # Calibration data is present
188 self.set_mode(0)
189 print("Found IMU calibration data, skipping calibration")
190 with open("IMU_cal.txt", "r") as file:
191 entries = file.readline().strip().split(",")
192 if len(entries) != 11:
193 raise ValueError("Calibration data length != expected length")
194
195 for idx, coeff_reg in enumerate(IMU.reg.calibration_coefficients):
196 self._write_reg(coeff_reg, entries[idx])
197
198 self.calibrated = True
199 self.set_mode(IMU.default_mode_set)
200 return True
201 else:
202 print("No IMU calibration data found")
203 self.set_mode(5) # set to NDOF mode for calibration
204
205 prev_cal = cur_cal = 0
206 while not (cur_cal & 0b0011_0011 == 0b0011_0011): # only check GYR and MAG calibration
207 if prev_cal != cur_cal:
208 print(f"{cur_cal:08b}")
209 prev_cal = cur_cal
210
211 cur_cal = self.get_calibration_status()[0] # Update cur_cal before checking
212 if cur_cal & 0b0011_0011 == 0b0011_0011:
213 self.set_mode(0)
214 with open("IMU_cal.txt", "w") as f:
215 coefs = ",".join(f"{s & 0xFFFF:016b}" for s in self.get_calibration_coefficients())
216 f.write(f"{coefs}\n")

235 def get_heading(self) -> float:
236 self.delta = self.get_imu_heading() - self.prev_imu_heading
237 self.prev_imu_heading = self.get_imu_heading()
238
239 # Overflow
240 if self.delta < -self.ar / 2:
241 self.delta += self.ar
242 # Underflow
243 if self.delta > self.ar / 2:
244 self.delta -= self.ar
245
246 self.heading += self.delta
247
248 return self.heading
