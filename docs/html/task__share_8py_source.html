173 def put(self, item, in_ISR=False):
174 # If we're in an ISR and the queue is full and we're not allowed to
175 # overwrite data, we have to give up and exit
176 if self.full():
177 if in_ISR:
178 return
179
180 # Wait (if needed) until there's room in the buffer for the data
181 if not self._overwrite:
182 while self.full():
183 pass
184
185 # Prevent data corruption by blocking interrupts during data transfer
186 if self._thread_protect and not in_ISR:
187 _irq_state = pyb.disable_irq()
188
189 # Write the data and advance the counts and pointers
190 self._buffer[self._wr_idx] = item
191 self._wr_idx += 1
192 if self._wr_idx >= self._size:
193 self._wr_idx = 0
194 self._num_items += 1
195 if self._num_items >= self._size: # Can't be fuller than full
196 self._num_items = self._size
197 if self._num_items > self._max_full: # Record maximum fillage
198 self._max_full = self._num_items
